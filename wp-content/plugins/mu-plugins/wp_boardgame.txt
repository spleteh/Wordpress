CREATE TABLE wp_boardgame(
ID INT( 11 ) NOT NULL ,
TITLE TEXT NOT NULL DEFAULT  '',
PLAYERS VARCHAR( 30 ) NOT NULL DEFAULT  '',
TIME INT( 4 ) NOT NULL DEFAULT  '0',
YEAR INT( 4 ) NOT NULL DEFAULT  '0',
PUBLISHER varchar(50) NOT NULL DEFAULT '',
ARTIST varchar(50) NOT NULL DEFAULT '',
DESIGNER varchar(50) NOT NULL DEFAULT '',
AUTHOR text NOT NULL DEFAULT '',
HONORS varchar(100) NOT NULL DEFAULT '',
PRIMARY KEY ( ID )
)
